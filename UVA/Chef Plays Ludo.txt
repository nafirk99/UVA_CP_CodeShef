using System;
					
public class Program
{
	public static void Main()
	{
		int T = int.Parse(Console.ReadLine());
		for (int i=1; i<=T; i++)
		{
			int dice = int.Parse(Console.ReadLine().Trim());
			Console.WriteLine(dice == 6 ? "YES" : "NO");
			
		}
	}
}
