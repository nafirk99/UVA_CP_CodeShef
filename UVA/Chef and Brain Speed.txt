using System;

public class Program
{
	public static void Main()
	{
		string T = Console.ReadLine();
		string[] t = T.Split(' ', ' ');
		int a = int.Parse(t[0]);
		int b = int.Parse(t[1]);
		if (b > a)
			Console.WriteLine("YES");
		else
			Console.WriteLine("NO");
	}
}
