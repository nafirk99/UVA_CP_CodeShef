using System;
using System.Text;

class Program
{
    static void Main()
    {
        Console.WriteLine("Enter the first very large number:");
        string num1 = Console.ReadLine();

        Console.WriteLine("Enter the second very large number:");
        string num2 = Console.ReadLine();

        string sum = AddVeryLargeNumbers(num1, num2);

        Console.WriteLine("Sum:");
        Console.WriteLine(sum);
    }

    static string AddVeryLargeNumbers(string num1, string num2)
    {
        StringBuilder result = new StringBuilder();

        int carry = 0;
        int maxLength = Math.Max(num1.Length, num2.Length);

        // Pad the numbers with leading zeros to make them of equal length
        num1 = num1.PadLeft(maxLength, '0');
        num2 = num2.PadLeft(maxLength, '0');

        // Perform addition digit by digit
        for (int i = maxLength - 1; i >= 0; i--)
        {
            int digit1 = num1[i] - '0';
            int digit2 = num2[i] - '0';

            int digitSum = digit1 + digit2 + carry;
            carry = digitSum / 10;
            result.Insert(0, digitSum % 10);
        }

        // If there's a carry left, append it to the result
        if (carry > 0)
        {
            result.Insert(0, carry);
        }

        return result.ToString();
    }
}
