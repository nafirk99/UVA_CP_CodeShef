using System;
					
public class Program
{
	public static void Main()
	{
		string T = Console.ReadLine();
		string[] R = T.Split(' ');
		int c = 0;
		int a = int.Parse(R[0]);
		int b = int.Parse(R[1]);

		{
			while(a<=b)
			{
		      a=a*3;
              b=b*2;
              c++;
			}
		}
		Console.WriteLine(c);
	}
}
