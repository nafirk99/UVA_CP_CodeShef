using System;
					
public class Program
{
	public static void Main()
	{
		string T = Console.ReadLine();
		int flag = 0;
		for(int i=0; i<T.Length; i++)
		{
			if(T[i]=='H' || T[i]=='Q' || T[i]=='9')
			{
				flag = 1;
			}
			//else { flag = 0; }
		}
		if (flag == 1){ Console.WriteLine("YES"); }
		else { Console.WriteLine("NO"); }
	}
}
